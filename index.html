class SMAStrategy(StrategyCore):
    """Simple Moving Average Cross Strategy"""
    
    def calculate_indicators(self, data):
        # Calculate fast and slow SMA
        sma_fast = self.calculate_sma(data, 10)
        sma_slow = self.calculate_sma(data, 30)
        return {'fast': sma_fast, 'slow': sma_slow}
    
    def generate_signals(self, indicators):
        # Generate buy/sell signals
        if indicators['fast'] > indicators['slow']:
            return 'buy'
        elif indicators['fast'] < indicators['slow']:
            return 'sell'
        return None

sma_strategy.py 
from quantcell.strategy import StrategyCore
from quantcell.backtest import StrategyRunner

class SMAStrategy(StrategyCore):
    """Dual SMA Cross Strategy"""
    
    def __init__(self):
        self.fast_period = 10
        self.slow_period = 30
    
    def calculate_indicators(self, data):
        return {
            'sma_fast': data.close.rolling(self.fast_period).mean(),
            'sma_slow': data.close.rolling(self.slow_period).mean()
        }
    
    def generate_signals(self, indicators):
        if indicators['sma_fast'] > indicators['sma_slow']:
            return 'buy'
        return 'sell'

# Run backtest
runner = StrategyRunner(SMAStrategy())
results = runner.run(data, initial_cash=10000)
print(f"Return: {results.return_pct:.2f}%")
